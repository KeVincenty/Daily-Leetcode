class StockSpanner:

    def __init__(self):
        self.stack = [(float('inf'), 0)]
        self.cur = 0

    def next(self, price: int) -> int:
        self.cur += 1
        while len(self.stack) != 0 and price >= self.stack[-1][0]:
            self.stack.pop()
        res = self.cur - self.stack[-1][1]
        self.stack.append((price, self.cur))
        return res
