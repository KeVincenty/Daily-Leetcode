```python
class Solution:
    def longestArithSeqLength(self, nums: List[int]) -> int:
        dp = {}
        for right in range(len(nums)):
            dp[(right, 0)] = 1
            for left in range(right):
                diff = nums[right] - nums[left]
                dp[(right, diff)] = dp.get((left, diff), 1) + 1
        return max(dp.values())
```
